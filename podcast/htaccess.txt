RewriteEngine On

# 1️⃣ If user visits /podcast/ → show podcast.html
DirectoryIndex podcast.html

# 2️⃣ Redirect /podcast/*.html → /podcast/*
# (But do NOT redirect podcast.html itself)
RewriteCond %{THE_REQUEST} \s/+podcast/([^/]+)\.html[\s?] [NC]
RewriteCond %1 !^podcast$        # prevent redirecting podcast.html → /podcast/
RewriteRule ^([^/]+)\.html$ $1 [R=301,L]

# 3️⃣ Serve /podcast/* → /podcast/*.html
RewriteCond %{REQUEST_FILENAME}.html -f
RewriteRule ^([^/]+)/?$ $1.html [L]
